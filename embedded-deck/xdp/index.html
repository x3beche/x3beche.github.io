00 XDP nedir?

XDP (eXpress Data Path), Linux kernel'ının NIC sürücüsünde — sk_buff tahsis edilmeden önce — eBPF programı çalıştırmasına olanak tanıyan yüksek performanslı paket işleme çerçevesidir. Kernel bypass değil, kernel-accelerated yaklaşımı.

01 XDP hook noktaları

XDP programı üç farklı modda bağlanabilir. Mod seçimi, performans ve donanım desteği arasındaki dengeyi belirler.

XDP modları

XDP_NATIVE (offload'sız en hızlı)NIC sürücüsünde, NAPI poll'dan önce çağrılır. sk_buff tahsis edilmez. En yüksek performans. Sürücü desteği gerektirir: mlx5, i40e, ixgbe, virtio, tun, veth, bonding vb.
XDP_GENERIC (herhangi sürücü)sk_buff oluşturulduktan sonra çağrılır — sürücü bağımsız, tüm NIC'lerde çalışır. Performans düşük; geliştirme ve test için kullanılır.
XDP_OFFLOAD (NIC'e offload)eBPF programı NIC donanımına yüklenir, CPU müdahalesi sıfır. Çok sınırlı NIC desteği (Netronome Agilio). En yüksek performans ama eBPF kısıtları çok daha katı.

Driver desteği kontrolü

# XDP native desteği olan NIC'leri listele
# ethtool ile sürücü bilgisi
ethtool -i eth0 | grep driver

# XDP programı bağla (native mod)
ip link set dev eth0 xdp obj xdp_prog.o sec xdp
# Eğer sürücü native desteklemiyorsa generic'e düşer

# Sadece native mod zorla — generic'e fallback yok
ip link set dev eth0 xdpdrv obj xdp_prog.o sec xdp

# Generic mod (geliştirme için)
ip link set dev eth0 xdpgeneric obj xdp_prog.o sec xdp

# XDP programını kaldır
ip link set dev eth0 xdp off

# Mevcut XDP programını göster
ip link show eth0 | grep xdp
bpftool net show dev eth0

02 BPF program yazımı

XDP programı, C dilinde yazılan ve LLVM/Clang ile BPF bytecode'a derlenen küçük kernel fonksiyonudur. struct xdp_md context'i paketin DMA tamponuna erişim sağlar.

Minimal XDP programı

#include <linux/bpf.h>
#include <bpf/bpf_helpers.h>

/* Her gelen paketi düşür */
SEC("xdp")
int xdp_drop_all(struct xdp_md *ctx) {
    return XDP_DROP;
}

/* Her gelen paketi geçir */
SEC("xdp_pass")
int xdp_pass_all(struct xdp_md *ctx) {
    return XDP_PASS;
}

xdp_md context ve paket erişimi

#include <linux/bpf.h>
#include <linux/if_ether.h>
#include <linux/ip.h>
#include <linux/tcp.h>
#include <bpf/bpf_helpers.h>
#include <bpf/bpf_endian.h>

SEC("xdp")
int xdp_parse_packet(struct xdp_md *ctx) {
    /* ctx: paket DMA tamponuna pointer */
    void *data     = (void *)(long)ctx->data;
    void *data_end = (void *)(long)ctx->data_end;

    /* BPF verifier: her pointer erişimi bounds check gerektiriyor */

    /* Ethernet header */
    struct ethhdr *eth = data;
    if ((void *)(eth + 1) > data_end)
        return XDP_PASS;   /* too short, pass to kernel */

    /* Yalnızca IPv4'e bak */
    if (eth->h_proto != bpf_htons(ETH_P_IP))
        return XDP_PASS;

    /* IP header */
    struct iphdr *iph = (void *)(eth + 1);
    if ((void *)(iph + 1) > data_end)
        return XDP_PASS;

    /* TCP mı? */
    if (iph->protocol != IPPROTO_TCP)
        return XDP_PASS;

    /* TCP header */
    struct tcphdr *tcph = (void *)iph + (iph->ihl * 4);
    if ((void *)(tcph + 1) > data_end)
        return XDP_PASS;

    /* 80 numaralı porta gelen trafiği düşür */
    if (tcph->dest == bpf_htons(80))
        return XDP_DROP;

    return XDP_PASS;
}

03 libbpf API

libbpf, BPF programlarını yüklemek, map'lere erişmek ve XDP'yi bir arayüze bağlamak için kullanılan C kütüphanesidir. CO-RE (Compile Once, Run Everywhere) desteği ile kernel versiyonlarına taşınabilir.

Temel libbpf kullanımı

#include <bpf/libbpf.h>
#include <net/if.h>
#include <linux/if_link.h>

int main(int argc, char **argv) {
    const char *ifname = argv[1];  /* örn: "eth0" */
    int ifindex = if_nametoindex(ifname);

    /* BPF object dosyasını aç */
    struct bpf_object *obj = bpf_object__open_file("xdp_prog.o", NULL);
    if (libbpf_get_error(obj)) {
        perror("bpf_object__open_file");
        return 1;
    }

    /* Kernel'a yükle (verifier + JIT) */
    if (bpf_object__load(obj)) {
        perror("bpf_object__load");
        return 1;
    }

    /* "xdp" section'ındaki programı bul */
    struct bpf_program *prog = bpf_object__find_program_by_name(obj, "xdp_parse_packet");
    int prog_fd = bpf_program__fd(prog);

    /* XDP programını arayüze bağla (native mod) */
    int flags = XDP_FLAGS_DRV_MODE;  /* XDP_FLAGS_SKB_MODE = generic */
    if (bpf_xdp_attach(ifindex, prog_fd, flags, NULL) < 0) {
        perror("bpf_xdp_attach");
        return 1;
    }

    printf("XDP program attached to %s (fd=%d)\n", ifname, prog_fd);

    /* Ctrl+C ile bağlantıyı kes */
    pause();

    bpf_xdp_detach(ifindex, flags, NULL);
    bpf_object__close(obj);
    return 0;
}

04 eBPF maps

BPF map'ler, kernel (XDP programı) ve user-space arasında paylaşılan veri yapılarıdır. Paket sayaçları, engel listeleri, akış tabloları ve konfigürasyon değerleri burada tutulur.

XDP programında map tanımı

#include <linux/bpf.h>
#include <bpf/bpf_helpers.h>

/* Per-CPU sayaç array — CPU başına lock yok */
struct {
    __uint(type, BPF_MAP_TYPE_PERCPU_ARRAY);
    __uint(max_entries, 1);
    __type(key, __u32);
    __type(value, __u64);
} pkt_counter SEC(".maps");

/* IP engel listesi — hash map */
struct {
    __uint(type, BPF_MAP_TYPE_HASH);
    __uint(max_entries, 65536);
    __type(key,   __u32);  /* IPv4 adresi */
    __type(value, __u8);   /* 1 = engel */
} blocklist SEC(".maps");

/* LRU hash: bağlantı takibi */
struct flow_key {
    __u32 src_ip, dst_ip;
    __u16 src_port, dst_port;
    __u8  protocol;
};

struct {
    __uint(type, BPF_MAP_TYPE_LRU_HASH);
    __uint(max_entries, 1 << 20);  /* 1M akış */
    __type(key,   struct flow_key);
    __type(value, __u64);          /* paket sayısı */
} flow_table SEC(".maps");

SEC("xdp")
int xdp_counter(struct xdp_md *ctx) {
    __u32 key = 0;
    __u64 *cnt = bpf_map_lookup_elem(&pkt_counter, &key);
    if (cnt) (*cnt)++;

    /* IP engel listesi kontrolu */
    void *data     = (void *)(long)ctx->data;
    void *data_end = (void *)(long)ctx->data_end;
    struct ethhdr *eth = data;
    if ((void *)(eth + 1) > data_end) return XDP_PASS;
    if (eth->h_proto != bpf_htons(ETH_P_IP)) return XDP_PASS;

    struct iphdr *iph = (void *)(eth + 1);
    if ((void *)(iph + 1) > data_end) return XDP_PASS;

    __u32 src = iph->saddr;
    if (bpf_map_lookup_elem(&blocklist, &src))
        return XDP_DROP;  /* engel listesinde — drop */

    return XDP_PASS;
}

05 AF_XDP — user-space zero-copy

AF_XDP, XDP_REDIRECT aracılığıyla paketleri user-space'e aktarmanın zero-copy yöntemidir. UMEM (User MEMory) üzerinde tanımlanan ring buffer'lar sayesinde kernel-userspace arasında kopya yoktur.

AF_XDP mimarisi

  NIC RX
    │
  XDP program → XDP_REDIRECT → xsk_map
                                    │
  ┌─────────────────────────────────▼───────────────────┐
  │  UMEM (User Memory — mmap ile paylaşılan büyük bellek) │
  │                                                       │
  │  Fill Ring      ← user: "bu adresi doldur"            │
  │  Completion Ring← user: "TX tamamlandı"               │
  │  RX Ring        → user: paketler burada               │
  │  TX Ring        ← user: "bu paketi gönder"            │
  └───────────────────────────────────────────────────────┘

AF_XDP socket oluşturma

#include <linux/if_xdp.h>
#include <xdp/xsk.h>
#include <sys/mman.h>

#define UMEM_SIZE   (64 * 1024 * 1024)   /* 64 MB UMEM */
#define FRAME_SIZE  XSK_UMEM__DEFAULT_FRAME_SIZE  /* 4096 */
#define NUM_FRAMES  (UMEM_SIZE / FRAME_SIZE)

struct xsk_socket_info {
    struct xsk_socket *xsk;
    struct xsk_umem   *umem;
    struct xsk_ring_prod fq;   /* fill queue */
    struct xsk_ring_cons cq;   /* completion queue */
    struct xsk_ring_cons rx;   /* rx queue */
    struct xsk_ring_prod tx;   /* tx queue */
    void  *umem_area;
};

struct xsk_socket_info *create_xsk_socket(const char *ifname, __u32 queue_id) {
    struct xsk_socket_info *xsk_info = calloc(1, sizeof(*xsk_info));

    /* UMEM için mmap */
    xsk_info->umem_area = mmap(NULL, UMEM_SIZE,
                                PROT_READ | PROT_WRITE,
                                MAP_PRIVATE | MAP_ANONYMOUS | MAP_HUGETLB, -1, 0);

    /* UMEM oluştur */
    struct xsk_umem_config umem_cfg = {
        .fill_size  = XSK_RING_PROD__DEFAULT_NUM_DESCS,
        .comp_size  = XSK_RING_CONS__DEFAULT_NUM_DESCS,
        .frame_size = FRAME_SIZE,
        .frame_headroom = 0,
    };
    xsk_umem__create(&xsk_info->umem, xsk_info->umem_area, UMEM_SIZE,
                     &xsk_info->fq, &xsk_info->cq, &umem_cfg);

    /* XDP socket oluştur */
    struct xsk_socket_config xsk_cfg = {
        .rx_size    = XSK_RING_CONS__DEFAULT_NUM_DESCS,
        .tx_size    = XSK_RING_PROD__DEFAULT_NUM_DESCS,
        .libbpf_flags = 0,
        .xdp_flags  = XDP_FLAGS_DRV_MODE,
        .bind_flags = XDP_ZEROCOPY,  /* sıfır kopya */
    };
    xsk_socket__create(&xsk_info->xsk, ifname, queue_id,
                       xsk_info->umem, &xsk_info->rx,
                       &xsk_info->tx, &xsk_cfg);

    return xsk_info;
}

RX döngüsü

void rx_loop(struct xsk_socket_info *xsk) {
    __u32 idx_rx = 0, idx_fq = 0;
    unsigned int rcvd;

    while (1) {
        /* NIC'ten alınan paket sayısı */
        rcvd = xsk_ring_cons__peek(&xsk->rx, 64, &idx_rx);
        if (!rcvd) {
            /* Kernel'ı yokla (busy-poll yerine poll() kullanılabilir) */
            if (xsk_ring_prod__needs_wakeup(&xsk->fq))
                recvfrom(xsk_socket__fd(xsk->xsk), NULL, 0, MSG_DONTWAIT, NULL, NULL);
            continue;
        }

        /* Her paketi işle */
        for (__u32 i = 0; i < rcvd; i++) {
            const struct xdp_desc *desc = xsk_ring_cons__rx_desc(&xsk->rx, idx_rx++);
            __u64 addr = desc->addr;
            __u32 len  = desc->len;

            /* Zero-copy: UMEM'den doğrudan oku */
            void *pkt = xsk_umem__get_data(xsk->umem_area, addr);

            /* Paket işle: pkt[0..len] */
            process_packet(pkt, len);
        }

        xsk_ring_cons__release(&xsk->rx, rcvd);

        /* Fill queue'ya frame adresleri geri ver */
        xsk_ring_prod__reserve(&xsk->fq, rcvd, &idx_fq);
        for (__u32 i = 0; i < rcvd; i++)
            *xsk_ring_prod__fill_addr(&xsk->fq, idx_fq++) = i * FRAME_SIZE;
        xsk_ring_prod__submit(&xsk->fq, rcvd);
    }
}

Bu bölümde

  • UMEM: mmap ile oluşturulan büyük bellek havuzu, kernel ile paylaşımlı
  • XDP_ZEROCOPY: kopya yok — paket UMEM'den doğrudan okunur
  • Fill ring: user → kernel; RX ring: kernel → user; paket adresleri paylaşılır

06 XDP uygulamaları

DDoS mitigation, load balancer ve paket mirror — üç temel XDP kullanım senaryosunun BPF program yapıları.

DDoS mitigation — DROP loop

#include <linux/bpf.h>
#include <linux/if_ether.h>
#include <linux/ip.h>
#include <linux/udp.h>
#include <bpf/bpf_helpers.h>
#include <bpf/bpf_endian.h>

struct {
    __uint(type, BPF_MAP_TYPE_LPM_TRIE);      /* CIDR engel listesi */
    __uint(max_entries, 10000);
    __type(key, struct bpf_lpm_trie_key);
    __type(value, __u64);                      /* düşürülen paket sayısı */
    __uint(map_flags, BPF_F_NO_PREALLOC);
} ddos_blocklist SEC(".maps");

SEC("xdp")
int xdp_ddos_mitigation(struct xdp_md *ctx) {
    void *data     = (void *)(long)ctx->data;
    void *data_end = (void *)(long)ctx->data_end;

    struct ethhdr *eth = data;
    if ((void *)(eth + 1) > data_end) return XDP_PASS;
    if (eth->h_proto != bpf_htons(ETH_P_IP)) return XDP_PASS;

    struct iphdr *iph = (void *)(eth + 1);
    if ((void *)(iph + 1) > data_end) return XDP_PASS;

    /* LPM trie: CIDR subnet eşleştirmesi */
    struct {
        __u32 prefixlen;
        __u32 data;
    } key = { .prefixlen = 32, .data = iph->saddr };

    __u64 *cnt = bpf_map_lookup_elem(&ddos_blocklist, &key);
    if (cnt) {
        __sync_fetch_and_add(cnt, 1);
        return XDP_DROP;
    }

    return XDP_PASS;
}

L4 Load Balancer — REDIRECT

/* Basit round-robin L4 load balancer */
struct backend_info {
    __u32 ip;
    __u8  mac[6];
};

struct {
    __uint(type, BPF_MAP_TYPE_ARRAY);
    __uint(max_entries, 8);    /* 8 backend */
    __type(key,   __u32);
    __type(value, struct backend_info);
} backends SEC(".maps");

struct {
    __uint(type, BPF_MAP_TYPE_PERCPU_ARRAY);
    __uint(max_entries, 1);
    __type(key,   __u32);
    __type(value, __u32);      /* round-robin counter */
} rr_counter SEC(".maps");

SEC("xdp")
int xdp_lb(struct xdp_md *ctx) {
    /* ... parse eth/ip/tcp ... */

    /* Round-robin backend seç */
    __u32 key = 0;
    __u32 *rr = bpf_map_lookup_elem(&rr_counter, &key);
    if (!rr) return XDP_PASS;

    __u32 idx = (*rr)++ % 8;
    struct backend_info *backend = bpf_map_lookup_elem(&backends, &idx);
    if (!backend) return XDP_PASS;

    /* Destination IP ve MAC'i değiştir */
    /* iph->daddr = backend->ip; checksum güncelle */
    /* eth->h_dest = backend->mac; */

    return XDP_TX;  /* paketi değiştirilmiş haliyle geri gönder */
}

Bu bölümde

  • LPM_TRIE: CIDR subnet eşleştirmesi — /16 /24 /32 engel kuralları
  • XDP_TX: paketi modifiye et ve aynı NIC'ten geri gönder
  • XDP_REDIRECT: bpf_redirect() ile başka NIC/CPU'ya yönlendir

08 Pratik: 10 Mpps paket sayacı

AF_XDP ile zero-copy UDP paket yakalama ve 10 Mpps hedefine ulaşan tam C örneği — XDP programı + user-space sayaç uygulaması.

XDP BPF programı (xdp_counter_kern.c)

#include <linux/bpf.h>
#include <linux/if_ether.h>
#include <linux/ip.h>
#include <linux/udp.h>
#include <bpf/bpf_helpers.h>
#include <bpf/bpf_endian.h>

/* PERCPU sayaçlar: paket sayısı ve byte toplamı */
struct pkt_stats {
    __u64 packets;
    __u64 bytes;
};

struct {
    __uint(type, BPF_MAP_TYPE_PERCPU_ARRAY);
    __uint(max_entries, 1);
    __type(key,   __u32);
    __type(value, struct pkt_stats);
} stats_map SEC(".maps");

/* AF_XDP socket map — paketleri user-space'e yönlendir */
struct {
    __uint(type, BPF_MAP_TYPE_XSKMAP);
    __uint(max_entries, 64);   /* max queue sayısı */
    __type(key,   __u32);
    __type(value, __u32);
} xsk_map SEC(".maps");

SEC("xdp")
int xdp_pkt_counter(struct xdp_md *ctx) {
    void *data     = (void *)(long)ctx->data;
    void *data_end = (void *)(long)ctx->data_end;
    __u32 len = data_end - data;

    /* Sayacı güncelle */
    __u32 key = 0;
    struct pkt_stats *stats = bpf_map_lookup_elem(&stats_map, &key);
    if (stats) {
        stats->packets++;
        stats->bytes += len;
    }

    /* UDP paketi ise AF_XDP'ye yönlendir */
    struct ethhdr *eth = data;
    if ((void *)(eth + 1) <= data_end &&
        eth->h_proto == bpf_htons(ETH_P_IP)) {
        struct iphdr *iph = (void *)(eth + 1);
        if ((void *)(iph + 1) <= data_end &&
            iph->protocol == IPPROTO_UDP) {
            /* Paketi queue 0'daki AF_XDP socket'e yönlendir */
            return bpf_redirect_map(&xsk_map, ctx->rx_queue_index, XDP_PASS);
        }
    }

    return XDP_PASS;
}

char _license[] SEC("license") = "GPL";

User-space AF_XDP sayaç (xdp_counter_user.c)

#include <stdio.h>
#include <stdlib.h>
#include <signal.h>
#include <time.h>
#include <xdp/xsk.h>
#include <bpf/libbpf.h>
#include "xdp_counter_kern.skel.h"

static volatile sig_atomic_t stop = 0;
void sig_handler(int sig) { stop = 1; }

int main(int argc, char **argv) {
    if (argc < 2) { fprintf(stderr, "Kullanım: %s <ifname>\n", argv[0]); return 1; }
    const char *ifname = argv[1];

    signal(SIGINT, sig_handler);
    signal(SIGTERM, sig_handler);

    /* BPF skeleton aç ve yükle */
    struct xdp_counter_kern *skel = xdp_counter_kern__open_and_load();
    if (!skel) { perror("skeleton open_and_load"); return 1; }

    /* AF_XDP soketi oluştur */
    struct xsk_socket_info *xsk = create_xsk_socket(ifname, 0);

    /* xsk_map'e AF_XDP socket fd'yi kaydet */
    int xsk_map_fd = bpf_map__fd(skel->maps.xsk_map);
    __u32 queue_id = 0;
    int xsk_fd = xsk_socket__fd(xsk->xsk);
    bpf_map_update_elem(xsk_map_fd, &queue_id, &xsk_fd, BPF_ANY);

    /* XDP programını arayüze bağla */
    int ifindex = if_nametoindex(ifname);
    struct bpf_link *link = bpf_program__attach_xdp(
        skel->progs.xdp_pkt_counter, ifindex);

    int stats_fd = bpf_map__fd(skel->maps.stats_map);
    __u64 last_pkts = 0;
    struct timespec ts_prev, ts_curr;
    clock_gettime(CLOCK_MONOTONIC, &ts_prev);

    printf("%-12s %-12s %-10s\n", "Paket/s", "Toplam", "Mbps");

    while (!stop) {
        /* AF_XDP RX döngüsü çalıştır */
        rx_loop_once(xsk);

        /* Her saniye istatistik yazdır */
        clock_gettime(CLOCK_MONOTONIC, &ts_curr);
        double elapsed = (ts_curr.tv_sec - ts_prev.tv_sec)
                       + (ts_curr.tv_nsec - ts_prev.tv_nsec) * 1e-9;

        if (elapsed >= 1.0) {
            struct pkt_stats total = {0};
            __u32 key = 0;
            struct pkt_stats percpu[256];
            bpf_map_lookup_elem(stats_fd, &key, percpu);
            for (int i = 0; i < libbpf_num_possible_cpus(); i++) {
                total.packets += percpu[i].packets;
                total.bytes   += percpu[i].bytes;
            }

            double pps  = (total.packets - last_pkts) / elapsed;
            double mbps = (total.bytes * 8) / elapsed / 1e6;
            printf("%-12.0f %-12llu %-10.1f\n", pps, total.packets, mbps);

            last_pkts = total.packets;
            ts_prev = ts_curr;
        }
    }

    bpf_link__destroy(link);
    xdp_counter_kern__destroy(skel);
    return 0;
}

10 Mpps hedefine ulaşmak için: (1) native XDP mod, (2) mlx5/i40e/ixgbe NIC, (3) isolcpus kernel parametresi ile XDP CPU'larını izole et, (4) RX/TX irq affinity ayarla, (5) hugepage ile UMEM tahsis et, (6) NUMA farkındalığı — NIC ve UMEM aynı NUMA node'da olmalı.

  • bpf_redirect_map + XSKMAP: paketi AF_XDP socket'e yönlendir
