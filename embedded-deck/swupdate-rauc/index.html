03 SWUpdate Lua handler

Lua handler, sw-description'da script type ile tanДұmlanan Г¶zel gГјncelleme mantДұДҹДұnДұ uygulamak iГ§in kullanДұlДұr. SWUpdate API'sine tam eriЕҹim saДҹlar.

post_install.lua вҖ” Г¶rnek handlerв—Ҹ
-- post_install.lua
local swupdate = require("swupdate")
local posix    = require("posix")

local INFO    = swupdate.INFO
local WARNING = swupdate.WARNING
local ERROR   = swupdate.ERROR

-- GГјncelleme Г¶ncesi doДҹrulama
function preinst(image)
    swupdate.notify(INFO, "Pre-install: " .. image.filename)

    -- Disk alanДұ kontrolГј (min 100 MB)
    local stat    = posix.statvfs("/")
    local free_mb = (stat.f_bavail * stat.f_bsize) / (1024 * 1024)
    if free_mb < 100 then
        swupdate.notify(ERROR, "Yetersiz disk: " .. free_mb .. " MB")
        return false
    end

    swupdate.notify(INFO, "Disk alanДұ yeterli: " .. free_mb .. " MB")
    return true
end

-- GГјncelleme sonrasДұ adДұmlar
function postinst(image)
    swupdate.notify(INFO, "Post-install baЕҹlДұyor")

    -- Bootloader ortam deДҹiЕҹkenini gГјncelle
    local rc = os.execute("fw_setenv boot_slot b")
    if rc ~= 0 then
        swupdate.notify(ERROR, "fw_setenv baЕҹarДұsДұz!")
        return false
    end

    os.execute("fw_setenv bootcount 0")

    -- GГјncelleme logunu kaydet
    local log = io.open("/data/update_log.txt", "a")
    if log then
        log:write(os.date() .. " вҖ” v2.1.0 kuruldu\n")
        log:close()
    end

    swupdate.notify(INFO, "Post-install tamamlandДұ")
    return true
end

return {
    pre  = preinst,
    post = postinst,
}

Г–zel C handler вҖ” FPGA bitstream yazma

fpga_handler.cв—Ҹ
#include "swupdate.h"
#include "handler.h"
#include "util.h"

/* FPGA bitstream yazma handler'Дұ */
static int fpga_handler(struct img_type *img,
                        void __attribute__((__unused__)) *data)
{
    int fd;

    TRACE("FPGA bitstream yazma: %s", img->fname);

    /* FPGA manager sysfs Гјzerinden yaz */
    fd = open("/sys/class/fpga_manager/fpga0/firmware", O_WRONLY);
    if (fd < 0) {
        ERROR("FPGA manager aГ§ДұlamadДұ: %s", strerror(errno));
        return -EFAULT;
    }

    long long written = copyfile(img, &fd, img->size, &img->offset,
                                  0, 0, NULL);
    close(fd);

    if (written < 0) {
        ERROR("FPGA yazma hatasДұ");
        return written;
    }

    INFO("FPGA bitstream baЕҹarДұyla yazДұldДұ: %lld byte", written);
    return 0;
}

/* BaЕҹlangДұГ§ta handler'Дұ kaydet */
__attribute__((constructor))
static void fpga_handler_init(void)
{
    register_handler("fpga", fpga_handler, IMAGE_HANDLER, NULL);
}

Bundle

Slot

06 RAUC bundle oluЕҹturma

RAUC bundle, imzalanmДұЕҹ bir manifest ve payload image'larДұndan oluЕҹur. X.509 sertifikalarДұ ile imzalanДұr; doДҹrulama cihazda yerleЕҹik CA sertifikasДұyla yapДұlДұr.

X.509 sertifika altyapДұsДұ

sertifika oluЕҹturmaв—Ҹ

    -out ca.cert.pem -days 3650 -nodes \
    -subj "/CN=RAUC CA/O=My Company/C=TR"

## Д°mzalama sertifikasДұ oluЕҹtur

    -out developer.csr.pem \
    -subj "/CN=RAUC Developer/O=My Company/C=TR"

## CA ile imzala
openssl x509 -req -in developer.csr.pem \
    -CA ca.cert.pem -CAkey ca.key.pem \
    -CAcreateserial -out developer.cert.pem -days 365

#!/usr/bin/env python3
import requests, argparse

def upload(server, bundle_path, version):
    auth = ("admin", "admin")
    base = f"{server}/rest/v1"

    ## 1. Software module oluЕҹtur
    sm = requests.post(f"{base}/softwaremodules",
        json=[{"name": f"firmware-{version}",
               "version": version, "type": "os"}],
        auth=auth).json()[0]
    sm_id = sm["id"]

    ## 2. Bundle dosyasДұnДұ yГјkle
    with open(bundle_path, "rb") as f:
        requests.post(f"{base}/softwaremodules/{sm_id}/artifacts",
            files={"file": (bundle_path, f, "application/octet-stream")},
            auth=auth)

    ## 3. Distribution set oluЕҹtur
    ds = requests.post(f"{base}/distributionsets",
        json=[{"name": f"firmware-{version}", "version": version,
               "type": "os", "modules": [{"id": sm_id}]}],
        auth=auth).json()[0]

    print(f"Bundle hazДұr вҖ” DS ID={ds['id']}  version={version}")
    return ds["id"]

if __name__ == "__main__":
    p = argparse.ArgumentParser()
    p.add_argument("--bundle")
    p.add_argument("--server")
    p.add_argument("--version")
    args = p.parse_args()
    upload(args.server, args.bundle, args.version)
