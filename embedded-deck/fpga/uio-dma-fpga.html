00 UIO (Userspace I/O) sürücü çerçevesi

UIO, periferalin I/O register haritası ve kesme yönetimini kullanıcı alanına taşıyan bir Linux kernel çerçevesidir. Kernel'e minimal bir stub sürücü yazılır; asıl sürücü mantığı güvenli bir kullanıcı alanı programında uygulanır.

UIO mimarisi

  Kullanıcı alanı uygulama
       │
       │  open("/dev/uio0")
       │  mmap() → register'lar
       │  read() → kesme bekleme
       │  write() → kesme etkinleştir/temizle
       ↓
  ─────────────── Kernel sınırı ───────────────
       ↓
  UIO çekirdeği (drivers/uio/uio.c)
       │
       │  uio_register_device()
       ↓
  Minimal platform stub sürücüsü
  (uio_pdrv_genirq.c veya özel)
       │
       ↓
  Donanım: FPGA AXI periferali

UIO'nun avantajları ve sınırlamaları

Avantaj: hızSistem çağrısı yokken doğrudan mmap bellekten register okuma/yazma — düşük gecikme
Avantaj: basitlikKernel modülü yazmak yerine C/C++/Python ile kullanıcı alanında sürücü; kolay debug
Avantaj: yeniden başlatma güvenliğiKullanıcı programı çöküse donanım zarar görmez; kernel kodu kararlı kalır
Sınırlama: DMABüyük DMA transferleri için ek mekanizma (VFIO, CMA, /dev/mem) gerekir
Sınırlama: güvenlikroot erişimi veya uio grubuna üyelik gerekir; üretimde ACL ile kısıtla

Kernel minimal stub: uio_pdrv_genirq

Özel bir kernel modülü yazmak gerekmez; uio_pdrv_genirq sürücüsü Device Tree'deki compatible = "generic-uio" node'larını otomatik olarak UIO cihazına dönüştürür.

DTS — UIO node tanımı●
my_uio_periph: my-periph@40000000 {
    compatible = "generic-uio", "ui_pdrv";
    reg = <0x40000000 0x10000>;     /* base adres, boyut */
    interrupts = <0 29 4>;          /* IRQ tanımı */
    interrupt-parent = <&gic>;
};
cmdline / modprobe — UIO sürücüsünü etkinleştir●
# Kernel parametresi (U-Boot'ta)
# setenv bootargs ... uio_pdrv_genirq.of_id="generic-uio"

# veya modprobe ile yükleme
modprobe uio_pdrv_genirq of_id="generic-uio"

# Cihaz oluşturuldu mu?
ls /dev/uio*
# → /dev/uio0

# UIO cihaz bilgisi
cat /sys/class/uio/uio0/name
cat /sys/class/uio/uio0/maps/map0/addr
cat /sys/class/uio/uio0/maps/map0/size

Bu bölümde

  • UIO: minimal kernel stub + kullanıcı alanı sürücü mantığı
  • uio_pdrv_genirq + compatible="generic-uio": sıfır kernel kodu
  • mmap: register erişimi; read: kesme bekleme; write: kesme temizle
  • Avantaj: hız + basitlik; Sınırlama: büyük DMA için ek mekanizma

01 /dev/uio0 — mmap ile register erişimi

/dev/uio0 cihaz dosyası üzerinden mmap çağrısı yaparak FPGA periferalinin register haritası doğrudan işlem adres uzayına eşlenir. Bu sayede register'lara pointer erişimi gibi yazılabilir.

Temel UIO mmap şablonu

uio_mmap.c — tam şablon●
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/mman.h>
#include <errno.h>
#include <string.h>

#define UIO_DEV      "/dev/uio0"
#define UIO_MAP_SZ_PATH "/sys/class/uio/uio0/maps/map0/size"

static size_t get_uio_map_size(const char *path)
{
    FILE *f = fopen(path, "r");
    if (!f) return 0;
    size_t size;
    fscanf(f, "0x%zx", &size);
    fclose(f);
    return size;
}

int main(void)
{
    int fd = open(UIO_DEV, O_RDWR | O_SYNC);
    if (fd < 0) {
        fprintf(stderr, "open %s: %s\n", UIO_DEV, strerror(errno));
        return EXIT_FAILURE;
    }

    /* map0 boyutunu sysfs'ten oku */
    size_t map_size = get_uio_map_size(UIO_MAP_SZ_PATH);
    if (!map_size) map_size = 0x10000;  /* fallback 64 KB */

    /* map0 → mmap offset = 0 * getpagesize() */
    volatile uint32_t *regs = mmap(NULL, map_size,
                                   PROT_READ | PROT_WRITE,
                                   MAP_SHARED, fd, 0 * getpagesize());
    if (regs == MAP_FAILED) {
        perror("mmap");
        close(fd);
        return EXIT_FAILURE;
    }

    /* Register erişimi */
    printf("ID register: 0x%08X\n", regs[0x00 / 4]);
    printf("STATUS:      0x%08X\n", regs[0x04 / 4]);

    /* Kontrol register'ına yaz */
    regs[0x08 / 4] = 0x01;   /* enable */

    munmap((void *)regs, map_size);
    close(fd);
    return EXIT_SUCCESS;
}

Birden fazla bellek bölgesi (map1, map2...)

Bir UIO cihazı birden fazla bellek bölgesine sahip olabilir. Her bölge ayrı bir mmap çağrısıyla eşlenir; offset = bölge_indeksi × getpagesize() olarak hesaplanır. Örneğin bir DMA denetleyicisinin register haritası map0'da, DMA tampon alanı map1'de olabilir.

Çoklu map — map1 erişimi●
/* map1 için offset = 1 * page_size */
volatile uint32_t *dma_buf = mmap(NULL, dma_size,
    PROT_READ | PROT_WRITE, MAP_SHARED,
    fd, 1 * getpagesize());   /* map1 */

Bu bölümde

  • mmap(fd, 0*pagesize): map0 eşlemesi; volatile uint32_t pointer ile register erişimi
  • Map boyutu: /sys/class/uio/uio0/maps/map0/size sysfs'ten okunur
  • Çoklu map: offset = N × pagesize; her bölge bağımsız mmap

02 UIO ile kesme yönetimi

UIO üzerinden kesme bekleme, /dev/uio0'ı read() ile okumakla gerçekleşir. Kernel, donanım kesmesi geldiğinde read()'i döndürür; dönen değer kesme sayacıdır.

Kesme bekleme ve temizleme döngüsü

uio_irq.c — UIO interrupt yönetimi●
#include <stdio.h>
#include <stdint.h>
#include <fcntl.h>
#include <unistd.h>
#include <poll.h>

int uio_wait_irq(int uio_fd, int timeout_ms)
{
    struct pollfd pfd = {
        .fd     = uio_fd,
        .events = POLLIN,
    };
    int ret = poll(&pfd, 1, timeout_ms);
    if (ret < 0)  { perror("poll"); return -1; }
    if (ret == 0) { return 0; }   /* zaman aşımı */

    /* Kesme sayacını oku (bu kesmeden önce gelenleri temizler) */
    uint32_t irq_count;
    read(uio_fd, &irq_count, sizeof(irq_count));
    return (int)irq_count;
}

void uio_enable_irq(int uio_fd)
{
    /* 1 yaz → kernel IRQ'yu yeniden etkinleştirir (unmask) */
    uint32_t enable = 1;
    write(uio_fd, &enable, sizeof(enable));
}

int main(void)
{
    int fd = open("/dev/uio0", O_RDWR);
    /* ... mmap regs ... */

    uio_enable_irq(fd);         /* IRQ etkinleştir */

    for (;;) {
        int cnt = uio_wait_irq(fd, 1000);  /* 1 sn timeout */
        if (cnt <= 0) continue;

        printf("IRQ #%d alındı\n", cnt);

        /* Donanım kesme bayrağını temizle (register'a yaz) */
        /* regs[IRQ_STATUS / 4] = 0x1; */

        uio_enable_irq(fd);     /* Bir sonraki kesme için unmask */
    }
    close(fd);
    return 0;
}
DİKKAT

UIO read() çağrısı, kesmeyi kernel seviyesinde mask eder. Bir sonraki read()'den önce write(fd, 1) ile unmask yapılmazsa sonraki donanım kesmesi asla gelmiyor gibi görünür. Uygulama döngüsü: wait_irq → işle → donanım bayrağını temizle → enable_irq.

Bu bölümde

  • read(uio_fd): bloklar ve kesme gelince döner — dönen değer kesme sayacı
  • write(uio_fd, 1): IRQ'yu unmask eder — zorunlu her kesme sonrası
  • poll() ile timeout: bloksuz bekleme — birden fazla FD izleme
  • Döngü: wait → işle → hw bayrağı temizle → unmask → döngü

03 AXI-DMA sürücüsü (xilinx-dma)

Xilinx AXI-DMA IP bloğu için Linux kernel sürücüsü drivers/dma/xilinx/xilinx_dma.c'dedir. Bu sürücü, DMA Engine API üzerinden kullanılır ve hem doğrudan mod hem scatter-gather modunu destekler.

AXI-DMA blok diyagramı

  PS DDR ←─────────────────────────────── AXI-DMA ───── PL İşlemci
            S2MM (Stream-to-Memory)          │
            AXI4 (HP port)                   │ MM2S (Memory-to-Stream)
                                             │ AXI4 (HP port)
  PS DDR ──────────────────────────────────→ AXI-DMA ──→ PL İşlemci

  AXI-DMA → AXI4 master (HP port) → DDR: yüksek bant genişliği
  AXI-DMA → AXI4-Stream: PL işleme hattına bağlantı

Device Tree — AXI-DMA

AXI-DMA DTS tanımı●
axi_dma_0: dma@40400000 {
    compatible = "xlnx,axi-dma-1.00.a";
    reg = <0x40400000 0x10000>;
    #dma-cells = <1>;
    xlnx,addrwidth = <32>;

    dma-channel@40400000 {
        compatible = "xlnx,axi-dma-mm2s-channel";
        interrupts = <0 29 4>;
        interrupt-parent = <&gic>;
        xlnx,datawidth = <32>;
        xlnx,device-id = <0>;
    };

    dma-channel@40400030 {
        compatible = "xlnx,axi-dma-s2mm-channel";
        interrupts = <0 30 4>;
        interrupt-parent = <&gic>;
        xlnx,datawidth = <32>;
        xlnx,device-id = <0>;
    };
};

DMA Engine API — kanal alma

Kernel sürücüsü — DMA Engine API (C)●
#include <linux/dmaengine.h>

struct dma_chan *tx_chan, *rx_chan;

/* TX (MM2S) kanalı al */
tx_chan = dma_request_chan(&pdev->dev, "axidma0");
if (IS_ERR(tx_chan)) {
    dev_err(&pdev->dev, "MM2S kanalı alınamadı\n");
    return PTR_ERR(tx_chan);
}

/* RX (S2MM) kanalı al */
rx_chan = dma_request_chan(&pdev->dev, "axidma1");

/* Transfer hazırlama */
struct dma_async_tx_descriptor *desc;
dma_addr_t dma_src = dma_map_single(&pdev->dev,
                                     src_buf, buf_size,
                                     DMA_TO_DEVICE);
desc = dmaengine_prep_slave_single(tx_chan, dma_src,
                                   buf_size, DMA_MEM_TO_DEV,
                                   DMA_PREP_INTERRUPT);
desc->callback = tx_callback;
dmaengine_submit(desc);
dma_async_issue_pending(tx_chan);

Bu bölümde

  • xilinx_dma.c: MM2S + S2MM kanalları; DMA Engine API ile erişim
  • dma_request_chan(): DT'deki kanal adıyla kanalı al
  • dmaengine_prep_slave_single(): tek parça transfer; DMA_MEM_TO_DEV / DMA_DEV_TO_MEM
  • dmaengine_submit() + dma_async_issue_pending(): transferi başlat

04 Coherent vs streaming DMA — FPGA için seçim

Linux DMA API iki temel model sunar: coherent (tutarlı) ve streaming. FPGA uygulamalarında her birinin avantajları farklıdır.

Coherent DMA

dma_alloc_coherent() ile ayrılan bellek hem CPU hem DMA için her zaman tutarlıdır (önbellek bypass veya hardware snoop). Yazılan veri doğrudan görülür, flush/invalidate gerekmez. Ancak bellek ayrımı pahalıdır ve küçük, sık erişilen tamponlar için uygundur.

Streaming DMA

dma_map_single() / dma_map_sg() ile eşlenen bellek, CPU ve DMA kullanımı arasında sahiplik değiştirilir. Her yönde önbellek sync gerekir. Büyük tek seferlik transferler için daha verimlidir.

ÖzellikCoherentStreaming
Önbellek syncGerekmiyor (otomatik)Manuel: dma_sync_*
Bellek ayrımıPahalı (önyükleme/setup)Ucuz (herhangi bellek)
Tekrar kullanımUzun yaşam tamponuHer transfer için
Tipik FPGA kullanımıDMA tanımlayıcıları, halka tamponuBüyük video/ADC blokları
Zynq ACP portuDoğal: L2 önbellek tutarlıHP portunda snoop gerekir
NOT

Zynq'de ACP portu (Accelerator Coherency Port) üzerinden erişen PL, doğal olarak L2 önbellek tutarlılığına sahiptir. Bu sayede dma_alloc_coherent() gereksiz hale gelebilir. Ancak ACP portu bant genişliği HP portlarından daha düşüktür; büyük transferler için HP0–HP3 + streaming DMA tercih edilir.

Bu bölümde

  • Coherent: dma_alloc_coherent — önbellek sync yok; descriptor/ring için
  • Streaming: dma_map_single/sg — büyük tek seferlik transferler için
  • Zynq ACP: doğal önbellek tutarlılık; HP0–HP3: yüksek bant streamin DMA

05 Scatter-gather DMA tanımlayıcıları

Scatter-gather (SG) DMA, birden fazla fiziksel olarak ardışık olmayan bellek parçasının tek bir DMA transfer zinciri olarak işlenmesini sağlar. Xilinx AXI-DMA, SG modu destekler.

SG tanımlayıcı yapısı (Xilinx AXI-DMA)

AXI-DMA SG tanımlayıcı (donanım formatı)●
/* Xilinx AXI-DMA SG Tanımlayıcı (16 byte hizalı) */
struct xilinx_sg_desc {
    uint32_t next_desc;     /* Sonraki tanımlayıcı PA (4 byte hizalı) */
    uint32_t reserved0;
    uint32_t buf_addr;      /* Transfer tampon fiziksel adresi */
    uint32_t reserved1;
    uint32_t control;       /* bit[26]: TXSOF, bit[25]: TXEOF, bit[22:0]: uzunluk */
    uint32_t status;        /* bit[31]: tamamlandı, bit[22:0]: transfer edilen byte */
    uint32_t app[5];        /* Uygulama alanları (isteğe bağlı) */
} __attribute__((packed, aligned(64)));   /* 64-byte hizalama zorunlu */

/* Control register bit tanımları */
#define XILINX_DMA_BD_SOP  (1 << 27)   /* Start of Packet */
#define XILINX_DMA_BD_EOP  (1 << 26)   /* End of Packet */

SG zinciri oluşturma (Linux DMA Engine)

dmaengine_prep_slave_sg — scatter-gather transfer●
#include <linux/scatterlist.h>

struct scatterlist sg[4];
sg_init_table(sg, 4);

/* Her parça için fiziksel adres ve uzunluk */
sg_set_buf(&sg[0], buf0, 4096);
sg_set_buf(&sg[1], buf1, 4096);
sg_set_buf(&sg[2], buf2, 4096);
sg_set_buf(&sg[3], buf3, 4096);

/* DMA adresi haritala */
int nents = dma_map_sg(&pdev->dev, sg, 4, DMA_TO_DEVICE);

/* SG transfer hazırla */
struct dma_async_tx_descriptor *desc;
desc = dmaengine_prep_slave_sg(tx_chan, sg, nents,
                               DMA_MEM_TO_DEV,
                               DMA_PREP_INTERRUPT);
desc->callback = sg_tx_done;
dmaengine_submit(desc);
dma_async_issue_pending(tx_chan);

/* Transfer sonrası haritayı kaldır */
dma_unmap_sg(&pdev->dev, sg, 4, DMA_TO_DEVICE);

Bu bölümde

  • SG: fiziksel olarak parçalı bellek → tek DMA zinciri
  • Xilinx BD: 64-byte hizalı; SOP/EOP bitleri paket sınırları
  • dma_map_sg → dmaengine_prep_slave_sg → submit → dma_unmap_sg
  • status[31]: tamamlama biti; DMA kesme callback'te kontrol edilir

Xilinx XDMA sürücüsü (PCIe tabanlı sistemlerde) ve ilgili kullanıcı alanı kütüphanesi, FPGA ile DDR arasında yüksek verimli DMA transferi sağlar. Zynq-bazlı sistemlerde alternatif UIO+CMA yaklaşımı kullanılır.

07 CMA ve dma_alloc_coherent — bellek ayırıcısı

CMA (Contiguous Memory Allocator), kernel çalışma sırasında büyük fiziksel olarak ardışık DMA tamponlarının ayrılmasını sağlar. dma_alloc_coherent(), bu havuzdan önbellek tutarlı bellek ayırır.

CMA yapılandırması

CMA boyut yapılandırması●
# Kernel command line (U-Boot bootargs)
# 128 MB CMA alanı rezerve et
setenv bootargs "... cma=128M"

# Veya Device Tree'de (reserved-memory):
reserved-memory {
    #address-cells = <1>;
    #size-cells    = <1>;
    ranges;

    cma_region: cma@1e000000 {
        compatible    = "shared-dma-pool";
        reusable;
        size          = <0x8000000>;   /* 128 MB */
        alignment     = <0x2000>;      /* 8 KB */
        linux,cma-default;
    };
};

dma_alloc_coherent kullanımı (kernel)

dma_alloc_coherent — kernel sürücüsünde●
#include <linux/dma-mapping.h>

#define DMA_BUF_SIZE  (4 * 1024 * 1024)   /* 4 MB */

dma_addr_t dma_handle;
void *cpu_addr;

/* Önbellek tutarlı DMA tamponu ayır */
cpu_addr = dma_alloc_coherent(&pdev->dev, DMA_BUF_SIZE,
                               &dma_handle, GFP_KERNEL);
if (!cpu_addr) {
    dev_err(&pdev->dev, "DMA tampon ayrılamadı\n");
    return -ENOMEM;
}

/* cpu_addr: CPU pointer (virtual)
   dma_handle: FPGA'nın kullanacağı fiziksel adres */
memset(cpu_addr, 0, DMA_BUF_SIZE);

/* FPGA AXI-DMA'ya fiziksel adresi ver */
iowrite32(dma_handle, dma_regs + AXI_DMA_SA);
iowrite32(DMA_BUF_SIZE, dma_regs + AXI_DMA_LENGTH);

/* Serbest bırakma */
dma_free_coherent(&pdev->dev, DMA_BUF_SIZE,
                  cpu_addr, dma_handle);

Bu bölümde

  • CMA: cma=128M cmdline veya reserved-memory DTS — ardışık fiziksel havuz
  • dma_alloc_coherent: cpu_addr (sanal) + dma_handle (fiziksel) döndürür
  • FPGA DMA motoru dma_handle adresini kullanır; CPU cpu_addr ile erişir
  • Önbellek sync yok: coherent bellek, otomatik tutarlı

08 Pratik: PS DDR ↔ FPGA AXI-DMA tam transfer programı

Bu bölümde kullanıcı alanından UIO + u-dma-buf kullanılarak PS DDR ile FPGA arasında tam bir AXI-DMA transfer döngüsü uygulanmaktadır. Örnek, MM2S ve S2MM kanallarını aynı anda kullanır.

Donanım yapılandırması (Vivado BD)

Vivado Block Design'da: Zynq PS + AXI-DMA (SG modu, 32-bit) + AXI-Stream passthrough (loopback). AXI-DMA'nın M_AXI_SG ve M_AXI_MM2S portları HP0'a, S_AXI_LITE portu GP0'a bağlı. IRQ'lar IRQ_F2P[0:1]'e bağlı.

Tam transfer programı (C)

axidma_loopback.c — tam çalışan program●
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <string.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/mman.h>
#include <poll.h>

/* AXI-DMA register ofsetleri */
#define DMA_MM2S_CR    0x00   /* MM2S Control */
#define DMA_MM2S_SR    0x04   /* MM2S Status */
#define DMA_MM2S_SA    0x18   /* MM2S Source Address */
#define DMA_MM2S_LEN   0x28   /* MM2S Transfer Length */
#define DMA_S2MM_CR    0x30   /* S2MM Control */
#define DMA_S2MM_SR    0x34   /* S2MM Status */
#define DMA_S2MM_DA    0x48   /* S2MM Dest Address */
#define DMA_S2MM_LEN   0x58   /* S2MM Transfer Length */

/* AXI-DMA Control register bitleri */
#define DMA_CR_RUN     (1 << 0)
#define DMA_CR_RESET   (1 << 2)
#define DMA_CR_IOC_IRQ (1 << 12)  /* Completion interrupt enable */

/* Parametreler */
#define DMA_BASE    0x40400000UL
#define DMA_MAP_SZ  0x10000UL
#define BUF_SIZE    (1024 * 1024)    /* 1 MB */

static volatile uint32_t *dma;   /* AXI-DMA register haritası */
static int uio_mm2s_fd, uio_s2mm_fd;

static uint64_t read_phys_addr(const char *path)
{
    FILE *f = fopen(path, "r");
    uint64_t addr = 0;
    if (f) { fscanf(f, "0x%lx", &addr); fclose(f); }
    return addr;
}

static void dma_write(int off, uint32_t val) { dma[off/4] = val; }
static uint32_t dma_read(int off)            { return dma[off/4]; }

int main(void)
{
    /* 1. AXI-DMA register'larını eşle (UIO0 = AXI-DMA) */
    uio_mm2s_fd = open("/dev/uio0", O_RDWR);   /* MM2S IRQ */
    uio_s2mm_fd = open("/dev/uio1", O_RDWR);   /* S2MM IRQ */
    int dma_fd  = open("/dev/uio2", O_RDWR);   /* DMA reg */

    dma = mmap(NULL, DMA_MAP_SZ, PROT_RW, MAP_SHARED, dma_fd, 0);

    /* 2. CMA tamponlarını al (u-dma-buf) */
    int src_fd = open("/dev/udmabuf0", O_RDWR);
    int dst_fd = open("/dev/udmabuf1", O_RDWR);

    uint8_t *src = mmap(NULL, BUF_SIZE, PROT_RW, MAP_SHARED, src_fd, 0);
    uint8_t *dst = mmap(NULL, BUF_SIZE, PROT_RW, MAP_SHARED, dst_fd, 0);

    uint64_t src_phys = read_phys_addr(
        "/sys/class/u-dma-buf/udmabuf0/phys_addr");
    uint64_t dst_phys = read_phys_addr(
        "/sys/class/u-dma-buf/udmabuf1/phys_addr");

    /* 3. Test verisi hazırla */
    for (int i = 0; i < BUF_SIZE; i++) src[i] = (uint8_t)(i & 0xFF);
    memset(dst, 0, BUF_SIZE);

    /* 4. DMA reset */
    dma_write(DMA_MM2S_CR, DMA_CR_RESET);
    dma_write(DMA_S2MM_CR, DMA_CR_RESET);
    usleep(1000);

    /* 5. Kesme etkinleştir ve çalıştır */
    dma_write(DMA_MM2S_CR, DMA_CR_RUN | DMA_CR_IOC_IRQ);
    dma_write(DMA_S2MM_CR, DMA_CR_RUN | DMA_CR_IOC_IRQ);

    /* 6. S2MM hedef adres + uzunluk (önce başlat) */
    dma_write(DMA_S2MM_DA,  (uint32_t)dst_phys);
    dma_write(DMA_S2MM_LEN, BUF_SIZE);

    /* 7. MM2S kaynak adres + uzunluk (transfer başlar) */
    dma_write(DMA_MM2S_SA,  (uint32_t)src_phys);
    dma_write(DMA_MM2S_LEN, BUF_SIZE);

    /* 8. Kesme bekle (poll) */
    struct pollfd pfds[2] = {
        { uio_mm2s_fd, POLLIN, 0 },
        { uio_s2mm_fd, POLLIN, 0 },
    };
    poll(pfds, 2, 5000);   /* 5 saniye timeout */

    uint32_t n;
    read(uio_mm2s_fd, &n, 4);
    read(uio_s2mm_fd, &n, 4);

    /* 9. Sonucu doğrula */
    int ok = (memcmp(src, dst, BUF_SIZE) == 0);
    printf("Transfer %s (%d KB)\n",
           ok ? "BAŞARILI" : "BAŞARISIZ", BUF_SIZE/1024);

    printf("MM2S SR: 0x%08X\n", dma_read(DMA_MM2S_SR));
    printf("S2MM SR: 0x%08X\n", dma_read(DMA_S2MM_SR));

    munmap(src, BUF_SIZE); close(src_fd);
    munmap(dst, BUF_SIZE); close(dst_fd);
    munmap((void*)dma, DMA_MAP_SZ); close(dma_fd);
    close(uio_mm2s_fd); close(uio_s2mm_fd);
    return ok ? 0 : 1;
}
İPUCU

DMA_S2MM_LEN yazmadan önce DMA_MM2S_LEN yazılırsa S2MM tamponu hazır olmadan transfer başlar; TLAST almadan önce S2MM durabilir. Güvenli sıra: S2MM_DA → S2MM_LEN → MM2S_SA → MM2S_LEN. Bu sayede veri alıcı hazır olduğunda gönderilir.

Bu bölümde

  • UIO: AXI-DMA register erişimi + IRQ bekleme; u-dma-buf: CMA tamponlar
  • Güvenli DMA sırası: RESET → S2MM_DA/LEN → MM2S_SA/LEN → poll IRQ
  • DMA_MM2S_SR / DMA_S2MM_SR: tamamlanma ve hata bitlerini kontrol et
  • memcmp loopback doğrulama: FPGA passthrough doğru çalışıyorsa eşleşmeli
