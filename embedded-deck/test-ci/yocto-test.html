00 Yocto Test Altyapısına Genel Bakış

Yocto/OpenEmbedded, image düzeyinde test için kapsamlı bir ekosistem sunar. Bu ekosistem birbirine bağlı birkaç katmandan oluşur.

  ┌──────────────────────────────────────────────────────────┐
  │                  YOCTO TEST EKOSİSTEMİ                   │
  ├────────────────┬─────────────────┬───────────────────────┤
  │   testimage    │     ptest       │     SDK Test          │
  │ (image-level) │  (package-level)│  (devtool test)       │
  ├────────────────┼─────────────────┼───────────────────────┤
  │ oeqa.runtime   │ ptest-runner    │  oeqa.sdk             │
  │ SSH executor   │ per-package     │  env setup            │
  │ QEMU / HW      │ test binary     │  cross-compile test   │
  └────────────────┴─────────────────┴───────────────────────┘

Test katmanları

testimage
Image seviyesi test. Tamamlanmış image'i QEMU'da boot eder, SSH ile bağlanır ve test case'leri çalıştırır. bitbake -c testimage myimage ile tetiklenir.
ptest
Paket seviyesi test. Her paket kendi test binary'sini içerir. ptest-runner hepsini çalıştırır. Yocto recipe'de inherit ptest gerekir.
oeqa.sdk
SDK test. Derlenmiş SDK ortamında derleme ve çalıştırma testleri yapar. bitbake -c testsdk myimage-sdk ile tetiklenir.
devtool test
Recipe geliştirirken kullanılan hızlı test döngüsü. devtool test myrecipe ile doğrudan çalıştırılır.

02 QEMU Üzerinde Test

Yocto testimage, gerçek donanım olmadan QEMU üzerinde tam image testi yapabilir. CI pipeline'ı için idealdir.

QEMU test konfigürasyonu

local.conf — QEMU test ayarları●
# QEMU için test hedefi
TEST_TARGET = "simcontroller"

# QEMU arch seçimi (MACHINE'den otomatik gelir ama override edilebilir)
# qemuarm64, qemuarm, qemux86_64, qemuriscv64
MACHINE = "qemuarm64"

# QEMU bellek miktarı
QB_MEM = "-m 1024"

# QEMU network (testimage SSH için gerekli)
QB_NETWORK_DEVICE = "-device virtio-net-device"

# KVM varsa kullan (CI host Linux ise)
# QB_MACHINE = "-machine virt,accel=kvm"

runqemu ile manuel test

bash — runqemu kullanımı●
# QEMU'yu başlat, nographic modda
runqemu qemuarm64 core-image-full-cmdline nographic

# Ağ erişimiyle birlikte (host'tan ssh ile bağlanmak için)
runqemu qemuarm64 core-image-full-cmdline nographic \
  slirp # TAP değil, kullanıcı modu network

# SSH bağlantısı (default port 2222)
ssh -p 2222 root@127.0.0.1

# runqemu içinde ek QEMU parametreleri
runqemu qemuarm64 core-image-full-cmdline nographic \
  "QB_KERNEL_CMDLINE_APPEND=systemd.unit=emergency.target"

QEMU KVM HIZLANDIRMA

GitLab CI runner'ınız KVM destekliyorsa: QB_MACHINE = "-machine virt,accel=kvm" ekleyerek QEMU hızını 5-10x artırabilirsiniz. Docker içinde KVM için --device /dev/kvm gerekir.

03 ptest Çerçevesi

ptest (package test), her Yocto paketinin kendi test binary'sini içermesini sağlar. ptest-runner tüm paket testlerini toplu olarak çalıştırır.

ptest nedir?

ptest, upstream projelerinin (openssl, glibc, bash, sqlite vs.) kendi test suite'lerini Yocto image'ına dahil etmenin standart yoludur. Bir recipe inherit ptest içerdiğinde, Yocto test binary'lerini ayrı bir paket (packagename-ptest) olarak derler.

ptest'i image'a eklemek

local.conf veya image recipe — ptest●
# ptest-runner ve seçili paket testlerini ekle
IMAGE_INSTALL:append = " ptest-runner"

# Hangi paketlerin ptest'i dahil edilsin?
DISTRO_FEATURES:append = " ptest"

# Belirli paketlerin ptest'leri:
IMAGE_INSTALL:append = " \
  openssl-ptest \
  glibc-ptest \
  sqlite3-ptest \
  python3-pytest-ptest \
  mylib-ptest \
"

# ptest test sonuç dizini
# /usr/lib/packagename/ptest/ altında test binary ve run-ptest scripti bulunur

ptest recipe örneği

mylib_1.0.bb — ptest recipe●
SUMMARY = "Örnek C kütüphanesi"
LICENSE = "MIT"
LIC_FILES_CHKSUM = "file://LICENSE;md5=..."

SRC_URI = "git://github.com/example/mylib.git;branch=main"
SRCREV = "abc1234"

# ptest sınıfını kullan
inherit ptest

# Normal derleme
do_compile() {
    oe_runmake
}

do_install() {
    install -d ${D}${libdir}
    install -m 0755 libmylib.so ${D}${libdir}/
    install -d ${D}${includedir}
    install -m 0644 mylib.h ${D}${includedir}/
}

# ptest binary ve run scripti yükle
do_install_ptest() {
    install -d ${D}${PTEST_PATH}
    # Test binary'sini kopyala
    install -m 0755 tests/test_mylib ${D}${PTEST_PATH}/
    # run-ptest scripti — ZORUNLU
    install -m 0755 ${WORKDIR}/run-ptest ${D}${PTEST_PATH}/
}

# ptest için bağımlılıklar
RDEPENDS:${PN}-ptest += "bash"

run-ptest script formatı

run-ptest — ptest çalıştırma scripti●
#!/bin/sh
# run-ptest — ptest-runner bu scripti çağırır
# Çıktı formatı: "PASS: test_name" veya "FAIL: test_name"

#!/bin/sh
BINDIR=$(dirname $0)

# Test binary'sini çalıştır
${BINDIR}/test_mylib | \
  sed -e 's/^ok /PASS: /' \
      -e 's/^not ok /FAIL: /' \
      -e 's/^# SKIP /SKIP: /'

Board üzerinde ptest çalıştırma

bash — ptest-runner kullanımı●
# Tüm ptest'leri çalıştır
ptest-runner

# Belirli paket testleri
ptest-runner openssl sqlite3 mylib

# Sonuçları dosyaya kaydet
ptest-runner 2>&1 | tee /tmp/ptest-results.log

# Çıktı formatı:
# START: ptest-runner
# BEGIN: /usr/lib/openssl/ptest/
# PASS: test_aes_128
# PASS: test_sha256
# FAIL: test_fips_mode
# END: /usr/lib/openssl/ptest/

04 Runtime Test Yazma — oeqa.runtime

Özel testimage test case'leri yazmak için oeqa.runtime.case.OERuntimeTestCase sınıfı kullanılır.

Özel test case yapısı

meta-custom/lib/oeqa/runtime/cases/myapp.py●
from oeqa.runtime.case import OERuntimeTestCase
from oeqa.core.decorator.depends import OETestDepends
from oeqa.core.decorator.oetimeout import OETimeout

class MyAppTest(OERuntimeTestCase):
    """myapp servisinin doğru çalıştığını test et"""

    @OETestDepends(['ssh.SSHTest.test_ssh'])
    def test_myapp_running(self):
        """myapp.service aktif mi?"""
        status, output = self.target.run('systemctl is-active myapp.service')
        self.assertEqual(status, 0, msg=f"myapp çalışmıyor: {output}")
        self.assertIn('active', output)

    @OETestDepends(['myapp.MyAppTest.test_myapp_running'])
    def test_myapp_version(self):
        """myapp versiyonu doğru mu?"""
        status, output = self.target.run('myapp --version')
        self.assertEqual(status, 0)
        self.assertIn('2.1', output)

    @OETimeout(60)
    def test_myapp_process_data(self):
        """myapp veri işleme testi"""
        # Dosyayı board'a kopyala
        self.target.copy_to('/tmp/test-input.bin', '/tmp/')
        status, output = self.target.run('myapp process /tmp/test-input.bin')
        self.assertEqual(status, 0)
        # Çıktı dosyasını host'a kopyala ve doğrula
        self.target.copy_from('/tmp/output.bin', '/tmp/')
        with open('/tmp/output.bin', 'rb') as f:
            data = f.read()
        self.assertEqual(data[:4], b'MYAP')  # magic header

    def test_myapp_config(self):
        """Konfigürasyon dosyası mevcut mu?"""
        status, _ = self.target.run('test -f /etc/myapp/myapp.conf')
        self.assertEqual(status, 0, "/etc/myapp/myapp.conf bulunamadı")

05 SDK Test

Yocto SDK'sı, cross-derleme ortamını test ederek geliştiricilerin doğru araç zincirine sahip olduğunu garantiler.

SDK derleme ve test

bash — SDK test●
# SDK derle
bitbake core-image-full-cmdline -c populate_sdk

# Çıktı: tmp/deploy/sdk/poky-*.sh

# SDK testlerini çalıştır
bitbake core-image-full-cmdline -c testsdk

# SDK'yı yükle ve test et
SDK="tmp/deploy/sdk/poky-glibc-x86_64-core-image-full-cmdline-aarch64-toolchain-4.0.sh"
chmod +x ${SDK}
${SDK} -y -d /opt/poky-sdk

# SDK ortamını başlat
source /opt/poky-sdk/environment-setup-aarch64-poky-linux

# Cross-compile testi
echo '#include <stdio.h>
int main(){printf("Hello from embedded\n");return 0;}' > /tmp/test.c
${CC} /tmp/test.c -o /tmp/test-aarch64
file /tmp/test-aarch64
# Çıktı: ELF 64-bit LSB executable, ARM aarch64

Özel SDK test case

meta-custom/lib/oeqa/sdk/cases/mylib.py●
import os, subprocess
from oeqa.sdk.case import OESDKTestCase

class MyLibSDKTest(OESDKTestCase):
    """mylib SDK testleri — cross-compile ve link doğrulama"""

    def test_header_exists(self):
        """mylib.h başlık dosyası SDK'da mevcut mu?"""
        header = os.path.join(self.tc.sdk_dir,
            "sysroots", self.tc.sdk_machine,
            "usr/include/mylib.h")
        self.assertTrue(os.path.exists(header),
            f"Başlık bulunamadı: {header}")

    def test_compile_and_link(self):
        """mylib ile basit bir program derle"""
        src = """
#include 
int main(void) {
    mylib_t *ctx = mylib_init();
    mylib_free(ctx);
    return 0;
}
"""
        with self._create_tmpdir() as tmpdir:
            srcfile = os.path.join(tmpdir, "test.c")
            with open(srcfile, "w") as f:
                f.write(src)
            result = subprocess.run(
                [f"{self.tc.env['CC']}", srcfile, "-lmylib", "-o",
                 os.path.join(tmpdir, "test")],
                env=self.tc.env,
                capture_output=True
            )
            self.assertEqual(result.returncode, 0,
                msg=result.stderr.decode())

07 Pratik: Custom ptest + testimage Pipeline

Sıfırdan bir custom ptest recipe yazmak ve GitLab CI'da tam testimage pipeline'ı çalıştırmak için adım adım rehber.

Adım 1: C kütüphanesi ve test kodu

mymath/mymath.c + tests/test_mymath.c●
/* mymath.c */
#include "mymath.h"
int mymath_add(int a, int b) { return a + b; }
int mymath_mul(int a, int b) { return a * b; }

/* tests/test_mymath.c — TAP protokolü kullanır */
#include <stdio.h>
#include "mymath.h"

static int pass=0, fail=0, total=0;

static void ok(int cond, const char *name) {
    total++;
    if (cond) { printf("ok %d - %s\n", total, name); pass++; }
    else       { printf("not ok %d - %s\n", total, name); fail++; }
}

int main(void) {
    printf("TAP version 13\n");
    ok(mymath_add(1, 2) == 3,   "add 1+2=3");
    ok(mymath_add(-1, 1) == 0,  "add -1+1=0");
    ok(mymath_mul(3, 4) == 12,  "mul 3*4=12");
    ok(mymath_mul(0, 99) == 0,  "mul 0*99=0");
    printf("1..%d\n", total);
    return fail > 0 ? 1 : 0;
}

Adım 2: Yocto recipe (ptest ile)

meta-custom/recipes-core/mymath/mymath_1.0.bb●
SUMMARY = "mymath — basit matematik kütüphanesi"
LICENSE = "MIT"
LIC_FILES_CHKSUM = "file://LICENSE;md5=abc123"

SRC_URI = "git://github.com/example/mymath.git;protocol=https;branch=main"
SRCREV  = "${AUTOREV}"

S = "${WORKDIR}/git"

inherit cmake ptest

EXTRA_OECMAKE = "-DBUILD_TESTING=ON"

do_install() {
    install -d ${D}${libdir}
    install -m 0755 ${B}/libmymath.so.1.0 ${D}${libdir}/libmymath.so.1.0
    ln -s libmymath.so.1.0 ${D}${libdir}/libmymath.so
    install -d ${D}${includedir}
    install -m 0644 ${S}/mymath.h ${D}${includedir}/
}

do_install_ptest() {
    install -d ${D}${PTEST_PATH}
    install -m 0755 ${B}/tests/test_mymath ${D}${PTEST_PATH}/
    install -m 0755 ${WORKDIR}/run-ptest  ${D}${PTEST_PATH}/
}

FILES:${PN} = "${libdir} ${includedir}"
RDEPENDS:${PN}-ptest = "bash"

Adım 3: run-ptest

meta-custom/recipes-core/mymath/files/run-ptest●
#!/bin/sh
cd $(dirname $0)
./test_mymath | sed -e 's/^ok /PASS: /' -e 's/^not ok /FAIL: /'

Adım 4: GitLab CI tam pipeline

stages:
  - yocto-build
  - qemu-test

    reports:

      TEST_TARGET = "simpleremote"

PIPELINE ÖZET

Bu pipeline şu adımları otomatik gerçekleştirir: Yocto build → QEMU testimage (ptest dahil) → QEMU geçerse gerçek RPi4'te wic image flash + testimage. Her aşama JUnit XML üretir ve GitLab test raporuna eklenir. Herhangi bir aşama başarısız olursa pipeline durur.
