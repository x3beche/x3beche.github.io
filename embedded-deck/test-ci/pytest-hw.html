00 HW-in-the-Loop Test Nedir?

HIL (Hardware-in-the-Loop), yazılımın simülatör yerine gerçek donanım üzerinde otomatik test edilmesi yaklaşımıdır.

Gömülü yazılım geliştirmede üç test katmanı vardır: birim testler (unit test — tamamen yazılım, simülasyon), entegrasyon testleri (gerçek donanım ile sınırlı), ve sistem testleri (tam sistem, gerçek donanım). HIL, ikinci ve üçüncü katmanı kapsar.

YaklaşımAvantajDezavantajNe yakalar?
QEMU/SimülatörHızlı, CI kolayDonanım davranışı eksikLogic hatalar
HIL (gerçek HW)Gerçek donanım davranışıYavaş, setup karmaşıkTiming, peripheral, güç hataları
Manuel testEsnekTekrarlanamaz, yavaşHer şey

Neden pytest?

pytest, gömülü donanım testi için ideal bir seçimdir çünkü:

  • Fixture sistemi, donanım kaynaklarının (serial port, JTAG, GPIO) setup/teardown'ını otomatik yönetir
  • @pytest.mark.parametrize ile aynı test birden fazla kart üzerinde kolayca çalıştırılır
  • JUnit XML çıktısı GitLab CI'a entegre edilir
  • Async test desteği (asyncio) timing testleri için güçlüdür
  • Zengin plugin ekosistemi (pytest-timeout, pytest-asyncio, pytest-html)
MİMARİ

HIL test runner (pytest) genellikle test host'ta (Linux PC veya CI runner) çalışır. DUT'a serial, USB-JTAG veya SSH üzerinden bağlanır. DUT güç kontrolü için PDU veya GPIO kullanılır. Board'un aynı ağda veya USB ile erişilebilir olması yeterlidir.

02 Serial Fixture — pyserial

pyserial, Python'dan UART serial konsola erişim sağlar. Gömülü boardların UART shell'ine programatik olarak komut göndermek için temel araçtır.

Temel serial fixture

fixtures/serial_fixture.py●
import pytest, serial, time, re

class SerialConsole:
    """UART konsol yardımcı sınıfı"""

    def __init__(self, port, baud=115200, timeout=30):
        self.ser = serial.Serial(
            port=port, baudrate=baud,
            timeout=timeout,
            bytesize=serial.EIGHTBITS,
            parity=serial.PARITY_NONE,
            stopbits=serial.STOPBITS_ONE
        )
        self.prompt = b"# "  # root shell prompt

    def cmd(self, command, timeout=30, expect=None):
        """Komut gönder, prompt'a kadar oku, çıktıyı döndür"""
        # Önceki bufferi temizle
        self.ser.reset_input_buffer()
        # Komutu gönder
        self.ser.write(command.encode() + b"\n")
        self.ser.flush()
        # Yanıt oku
        output = b""
        deadline = time.monotonic() + timeout
        while time.monotonic() < deadline:
            chunk = self.ser.read(self.ser.in_waiting or 1)
            output += chunk
            if self.prompt in output:
                break
        if expect and expect not in output:
            raise AssertionError(
                f"Beklenen çıktı '{expect}' bulunamadı.\nAlınan: {output!r}"
            )
        return output.decode(errors="replace")

    def wait_for(self, pattern, timeout=60):
        """Regex pattern eşleşmesini bekle"""
        buf = b""
        deadline = time.monotonic() + timeout
        while time.monotonic() < deadline:
            buf += self.ser.read(self.ser.in_waiting or 1)
            if re.search(pattern.encode(), buf):
                return buf.decode(errors="replace")
        raise TimeoutError(f"Pattern '{pattern}' {timeout}s içinde bulunamadı")

    def close(self): self.ser.close()


@pytest.fixture(scope="session")
def serial_console(board_config):
    console = SerialConsole(board_config["port"], board_config["baud"])
    yield console
    console.close()

03 JTAG / OpenOCD Fixture

OpenOCD, JTAG/SWD üzerinden flash, halt, resume ve bellek okuma işlemlerini Python'dan kontrol etmenizi sağlar.

OpenOCD RPC fixture

fixtures/jtag_fixture.py●
import pytest, socket, subprocess, time, os

class OpenOCDClient:
    """OpenOCD TCL RPC client"""
    SEPARATOR = b"\x1a"

    def __init__(self, host="localhost", port=6666):
        self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.sock.connect((host, port))
        self.sock.settimeout(10)

    def send(self, cmd):
        self.sock.send(cmd.encode() + self.SEPARATOR)
        resp = b""
        while not resp.endswith(self.SEPARATOR):
            resp += self.sock.recv(4096)
        return resp[:-1].decode(errors="replace").strip()

    def halt(self):
        self.send("halt")
        time.sleep(0.1)

    def resume(self):
        self.send("resume")

    def reset(self):
        self.send("reset halt")
        time.sleep(0.5)

    def flash(self, image_path, offset=0):
        """ELF veya bin dosyasını flash'a yaz"""
        self.halt()
        self.send(f"program {image_path} verify reset")
        time.sleep(2)  # flash yazma + verify bekle

    def read_mem32(self, addr):
        """32-bit bellek oku"""
        resp = self.send(f"mdw 0x{addr:08x}")
        # Örnek çıktı: "0x40000000: deadbeef"
        return int(resp.split(":")[1].strip(), 16)

    def close(self): self.sock.close()


@pytest.fixture(scope="session")

04 GPIO Fixture — gpiod

libgpiod Python binding'i, test host üzerindeki GPIO pinlerini kontrol ederek DUT ile doğrudan sinyal düzeyinde iletişim sağlar.

GPIO fixture

fixtures/gpio_fixture.py●
import pytest, gpiod, time

class GPIOController:
    """Test host GPIO kontrolcüsü (DUT reset, power, sinyal)"""

    def __init__(self, chip="gpiochip0"):
        self.chip = gpiod.Chip(chip)
        self.lines = {}

    def configure_output(self, pin, name, initial=0):
        line = self.chip.get_line(pin)
        config = gpiod.LineRequest()
        config.consumer = "hil-test"
        config.request_type = gpiod.LineRequest.DIRECTION_OUTPUT
        config.default_val = initial
        line.request(config)
        self.lines[name] = line
        return line

    def configure_input(self, pin, name):
        line = self.chip.get_line(pin)
        config = gpiod.LineRequest()
        config.consumer = "hil-test"
        config.request_type = gpiod.LineRequest.DIRECTION_INPUT
        line.request(config)
        self.lines[name] = line
        return line

    def reset_dut(self, reset_pin_name="dut_reset", pulse_ms=100):
        """DUT reset pulse gönder"""
        pin = self.lines[reset_pin_name]
        pin.set_value(0)  # active-low reset assert
        time.sleep(pulse_ms / 1000)
        pin.set_value(1)  # reset deassert

    def wait_for_signal(self, input_pin_name, value=1, timeout=10):
        """GPIO sinyal bekleme"""
        pin = self.lines[input_pin_name]
        deadline = time.monotonic() + timeout
        while time.monotonic() < deadline:
            if pin.get_value() == value:
                return True
            time.sleep(0.001)
        raise TimeoutError(f"GPIO '{input_pin_name}' = {value} beklendi, timeout")

    def close(self):
        for line in self.lines.values():
            line.release()
        self.chip.close()


@pytest.fixture(scope="session")
def gpio():
    ctrl = GPIOController("gpiochip0")
    # Test host GPIO pin atamaları
    ctrl.configure_output(17, "dut_reset", initial=1)   # GPIO17 → DUT RESET
    ctrl.configure_output(27, "dut_power", initial=1)   # GPIO27 → DUT PWR EN
    ctrl.configure_input(22,  "dut_status")              # GPIO22 ← DUT STATUS
    yield ctrl
    ctrl.close()

conftest.py — multi-board●
import pytest
from pathlib import Path
import yaml

# boards.yaml'dan konfigürasyonları yükle
BOARDS_CONFIG = yaml.safe_load(Path("boards.yaml").read_text())

def pytest_addoption(parser):
    parser.addoption("--board", action="append",
                     help="Test edilecek board(lar)")

def pytest_configure(config):
    config.addinivalue_line("markers",
        "board(name): sadece belirtilen board için çalıştır")

def pytest_collection_modifyitems(config, items):
    """--board filtresi uygula"""
    selected_boards = config.getoption("--board") or []
    if not selected_boards:
        return
    for item in items:
        board_marks = [m.args[0] for m in item.iter_markers("board")]
        if board_marks and not any(b in selected_boards for b in board_marks):
            item.add_marker(pytest.mark.skip(reason="board filtresi"))

@pytest.fixture(params=["rpi4", "imx8mp", "beagle"],
               ids=["rpi4", "imx8mp", "beagle"])
def any_board(request):
    """Tüm board'larda çalışacak fixture"""
    board_name = request.param
    cfg = BOARDS_CONFIG[board_name]
    return cfg

import pytest, asyncio, time
import asyncserial

@pytest.fixture(scope="session")
async def async_serial(board_config):
    ser = asyncserial.AsyncSerial(board_config["port"],
                                  baudrate=board_config["baud"])
    yield ser
    ser.close()

@pytest.mark.asyncio
async def test_uart_echo_latency(async_serial):
    """UART echo gidiş-dönüş süresi 10ms altında olmalı"""
    msg = b"PING\n"
    await async_serial.write(msg)
    t0 = time.perf_counter()
    resp = await asyncio.wait_for(
        async_serial.read_until(b"\n"), timeout=5.0
    )
    latency_ms = (time.perf_counter() - t0) * 1000
    assert b"PONG" in resp, f"Echo yanıtı yanlış: {resp}"
    assert latency_ms < 10, f"Latency çok yüksek: {latency_ms:.2f}ms"

@pytest.mark.asyncio
async def test_gpio_response_time(async_serial, gpio):
    """GPIO tetiklemesinden serial çıktıya kadar <1ms"""
    t0 = time.perf_counter()
    gpio.lines["dut_status"]  # tetikle — bu satır örnek amaçlı
    gpio.lines["dut_reset"].set_value(0)
    gpio.lines["dut_reset"].set_value(1)
    resp = await asyncio.wait_for(
        async_serial.read_until(b"BOOT OK\n"), timeout=5.0
    )
    elapsed_ms = (time.perf_counter() - t0) * 1000
    print(f"  GPIO → serial yanıt: {elapsed_ms:.2f}ms")
    assert elapsed_ms < 1000

test_gpio_toggle.py●
def test_gpio_output_toggle(serial_console, gpio):
    """DUT GPIO çıkışını test et — host tarafında sinyal ölç"""
    # DUT'a GPIO pin'i toggle et komutu ver
    serial_console.cmd("gpioset gpiochip0 5=1")
    time.sleep(0.01)
    val_high = gpio.lines["dut_status"].get_value()
    assert val_high == 1, "GPIO high beklendi"

    serial_console.cmd("gpioset gpiochip0 5=0")
    time.sleep(0.01)
    val_low = gpio.lines["dut_status"].get_value()
    assert val_low == 0, "GPIO low beklendi"

def test_gpio_toggle_frequency(serial_console, gpio):
    """1kHz GPIO toggle doğruluğu — 1000 kenar ölç"""
    serial_console.cmd("python3 -c \\"
        "import gpiod, time; c=gpiod.Chip('gpiochip0');"
        "l=c.get_line(5); [l.set_value(i%2) or time.sleep(0.0005) for i in range(2000)]\\"
        " &", timeout=1)
    # Host tarafında kenar sayısını ölç
    edges = 0
    t0 = time.monotonic()
    last_val = gpio.lines["dut_status"].get_value()
    while time.monotonic() - t0 < 1.0:
        val = gpio.lines["dut_status"].get_value()
        if val != last_val:
            edges += 1
            last_val = val
    # 1kHz = 2000 kenar/saniye, ±20% tolerans
    assert 1600 <= edges <= 2400, f"Kenar sayısı: {edges}, beklenen ~2000"

Test 3: Power cycle testi

test_power_cycle.py●
import pytest, time

@pytest.mark.destructive
@pytest.mark.slow
def test_power_cycle_recovery(serial_console, gpio):
    """DUT power cycle sonrası doğru boot edip ediyor?"""
    # 1. Mevcut durumu doğrula
    out = serial_console.cmd("uname -r")
    assert "6.1" in out

    # 2. Power cycle uygula
    gpio.lines["dut_power"].set_value(0)  # güç kes
    time.sleep(2)                          # kapasitörler için bekle
    gpio.lines["dut_power"].set_value(1)  # güç ver

    # 3. Boot tamamlanana kadar bekle
    serial_console.wait_for(
        pattern=r"root@[a-zA-Z0-9\-]+:~#",
        timeout=90
    )

    # 4. Boot sonrası doğrulama
    out = serial_console.cmd("uname -r")
    assert "6.1" in out, "Power cycle sonrası yanlış kernel"

    out = serial_console.cmd("systemctl is-system-running", timeout=15)
    assert out.strip() in ("running", "degraded"), \
        f"Sistem durumu: {out.strip()}"

@pytest.mark.parametrize("cycle_count", [5, 10, 50])
@pytest.mark.slow
def test_repeated_power_cycles(serial_console, gpio, cycle_count):
    """N kez power cycle sonrası sistem stabil mi?"""
    for i in range(cycle_count):
        gpio.lines["dut_power"].set_value(0)
        time.sleep(1)
        gpio.lines["dut_power"].set_value(1)
